API Endpoints

POST https://venonroblox-api.onrender.com/api/validate
Body: {"apiKey": "your_key", "userId": "robloxUserId", "productId": "your_product_id"}
GET https://venonroblox-api.onrender.com/api/user/{robloxUserId}/purchases
Header: X-API-Key: your_api_key
GET https://venonroblox-api.onrender.com/api/products/{productId}

Roblox Lua Example

local HttpService = game:GetService("HttpService")
local MarketplaceService = game:GetService("MarketplaceService")

local API_URL = "https://venonroblox-api.onrender.com/api/validate"
local API_KEY = "YOUR_API_KEY_HERE"

-- GamePass purchase handler
MarketplaceService.ProcessReceipt = function(receiptInfo)
    local userId = receiptInfo.PlayerId
    local productId = receiptInfo.ProductId
    
    -- Validate with Venon API
    local success, response = pcall(function()
        return HttpService:PostAsync(API_URL, HttpService:JSONEncode({
            apiKey = API_KEY,
            userId = tostring(userId),
            productId = productId
        }), Enum.HttpContentType.ApplicationJson)
    end)
    
    if success then
        local data = HttpService:JSONDecode(response)
        if data.valid then
            -- Give product to player
            giveProductToPlayer(userId, productId)
            return Enum.ProductPurchaseDecision.PurchaseGranted
        end
    end
    
    return Enum.ProductPurchaseDecision.NotProcessedYet
end

function giveProductToPlayer(userId, productId)
    -- Your logic to give the product
    print("Giving product", productId, "to user", userId)
end
